def create_lagged_features(data, column, max_lag):
    for lag in range(1, max_lag + 1):
        data[f'{column}_lag{lag}'] = data[column].shift(lag)
    return data

# Define the maximum number of lags
max_lag = 3

# Create lagged features for Vader and ChatGPT sentiment scores
